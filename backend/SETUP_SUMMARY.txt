npm start
   ↓
src/index.js (запуск сервера на порту 3000)
   ↓
src/app.js (конфигурация Express)
   ├─ подключает middleware (express.json)
   ├─ подключает swagger UI
   ├─ подключает маршруты
   └─ устанавливает обработчики ошибок
   ↓
src/routes/
   ├─ users.js (CRUD пользователей)
   ├─ products.js (CRUD продуктов)
   └─ все отправляют JSON ответы
   ↓
Клиент получает JSON ответ

�� HTTP МЕТОДЫ (CRUD ОПЕРАЦИИ):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

GET    = READ (прочитать)      получить данные
POST   = CREATE (создать)      создать новые данные
PUT    = UPDATE (обновить)     изменить данные
DELETE = DELETE (удалить)      удалить данные

📊 СТРУКТУРА JSON ОТВЕТА:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Успешный ответ (список):
{
  "success": true,
  "data": [...],
  "count": 3
}

Успешный ответ (один элемент):
{
  "success": true,
  "message": "Пользователь успешно создан",
  "data": { "id": 1, "name": "Иван", ... }
}

Ошибка:
{
  "success": false,
  "error": "Пользователь не найден"
}
